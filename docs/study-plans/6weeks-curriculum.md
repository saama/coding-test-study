Queue<int[]> queue = new LinkedList<>();
boolean[][] visited = new boolean[n][m];
int[] dx = {-1, 1, 0, 0};
int[] dy = {0, 0, -1, 1};

// DFS 템플릿  
void dfs(int x, int y) {
    visited[x][y] = true;
    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        // 범위 체크 + 방문 체크 + 조건 체크
    }
}
